Synopsis¶

  • Create/Delete Vpc and Vswitch, Querying Vswitch and VRouter and Adding Route Entry

Requirements (on host that executes module)¶

  • python >= 2.7
  • aliyun-python-sdk-core, aliyun-python-sdk-ecs, ecsutils and footmark

                                    Examples¶

                                    #
# provisioning to create vpc in VPC
#

# basic provisioning example to create vpc in VPC
- name: create vpc
  hosts: localhost
  connection: local
  vars:
    acs_access_key: xxxxxxxxxx
    acs_secret_access_key: xxxxxxxxxx
    region: cn-hongkong
    status: present
    cidr_block: 192.168.0.0/16
    vpc_name: Demo_VPC
    description: Demo VPC
    vswitches:
      - zone_id: 'cn-hongkong-b'
        description: 'dummy'
        cidr_block: '172.16.0.0/24'
  tasks:
    - name: create vpc
      ecs_vpc:
        acs_access_key: '{{ acs_access_key }}'
        acs_secret_access_key: '{{ acs_secret_access_key }}'
        region: '{{ region }}'
        status: '{{ status }}'
        cidr_block: '{{ cidr_block }}'
        vpc_name: '{{ vpc_name }}'
        description: '{{ description }}'
        vswitches: '{{ vswitches }}'
      register: result
    - debug: var=result
